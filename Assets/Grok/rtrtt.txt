public static class UniqueCombinationGenerator
{
    public static int GenerateUniqueNumber(int[] inputArray)
    {
        if (inputArray == null || inputArray.Length == 0)
            return 0;

        // Sort array to ensure consistent ordering
        Array.Sort(inputArray);
        
        // Count frequencies of each number
        Dictionary<int, int> frequencyMap = new();
        foreach (int num in inputArray)
        {
            frequencyMap[num] = frequencyMap.GetValueOrDefault(num) + 1;
        }

        // Calculate total combinations using multinomial coefficient
        int totalCombinations = CalculateMultinomialCoefficient(inputArray.Length, frequencyMap.Values);

        // Create a unique number based on the array's position in all possible combinations
        int result = 0;
        int remainingLength = inputArray.Length;
        
        // Process array from left to right
        for (int i = 0; i < inputArray.Length - 1; i++)
        {
            int currentNum = inputArray[i];
            remainingLength--;
            
            // Update frequency map for remaining elements
            Dictionary<int, int> remainingFreq = new(frequencyMap);
            remainingFreq[currentNum]--;
            if (remainingFreq[currentNum] == 0)
                remainingFreq.Remove(currentNum);

            // Count numbers less than currentNum that could be at this position
            int numbersBefore = 0;
            foreach (var kvp in remainingFreq.OrderBy(x => x.Key))
            {
                if (kvp.Key < currentNum)
                {
                    // Temporarily adjust frequency for this calculation
                    remainingFreq[kvp.Key]--;
                    if (remainingFreq[kvp.Key] == 0)
                        remainingFreq.Remove(kvp.Key);
                    
                    numbersBefore += CalculateMultinomialCoefficient(
                        remainingLength, 
                        remainingFreq.Values
                    );
                    
                    // Restore frequency
                    remainingFreq[kvp.Key] = kvp.Value;
                }
                else
                    break;
            }
            
            result += numbersBefore;
            frequencyMap[currentNum]--;
            if (frequencyMap[currentNum] == 0)
                frequencyMap.Remove(currentNum);
        }

        return result % totalCombinations; // Ensure result stays within bounds
    }

    private static int CalculateMultinomialCoefficient(int n, IEnumerable<int> frequencies)
    {
        // Calculate n! / (k1! * k2! * ... * km!)
        int numerator = Factorial(n);
        int denominator = 1;
        
        foreach (int freq in frequencies)
        {
            denominator *= Factorial(freq);
        }
        
        return numerator / denominator;
    }

    private static int Factorial(int n)
    {
        if (n <= 1) return 1;
        int result = 1;
        for (int i = 2; i <= n; i++)
        {
            result *= i;
        }
        return result;
    }
}
